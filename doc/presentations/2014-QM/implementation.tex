\myframe{proving things about \tla\ specs: \hfill proving leads-to}
{
\hspace{-18pt} {\color{Maroon}There are three rules that `make' a $\Diamond$:}
\vspace{15pt}     
\footnotesize
\begin{rl}[$\kw{WF}$ -- and similarly for $\kw{SF}$]
\begin{prooftree}
         \AxiomC{$I, I', P, [N]_V \seq P' \vee Q'$}
         \AxiomC{$I, I', P, \langle N \wedge A\rangle_V \seq  Q'$}
         \AxiomC{$I, P \seq \enabled\ \langle A \rangle_V$}
        % \RightLabel{\rlable{\kw{WF}}}    
         \TrinaryInfC{$\Box \,I,  \Box\, [N]_V, \kw{WF}_V(A)  \seq \Box( P \Rightarrow \Diamond Q)$}   
\end{prooftree}
\end{rl}

\vspace{20pt}

\begin{rl}[Lattice]
\begin{prooftree}
        \AxiomC{$(D, \succ) \ well-founded$}
        \AxiomC{$F, d \in D \seq \Box\big( G \Rightarrow \Diamond( H \vee \exists e \in D. \ d \succ e \wedge G[e/d] ) \big)  $}
%         \RightLabel{\rlable{Nec}}    
        \BinaryInfC{$ F, d \in D \seq  \Box(G \Rightarrow \Diamond H) $}
\end{prooftree}
\scriptsize
\hfill $d$ not in $H$, $e$ not in $G$
\end{rl}
\vspace{45pt}

}
